import json
import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
from io import BytesIO
import base64

try:
    # === TUNABLE PARAMETERS ===
    MAX_DEPTH = 5
    MIN_SAMPLES_SPLIT = 2
    RANDOM_STATE = 42
    TEST_SIZE = 0.2

    from sklearn.datasets import make_classification
    from sklearn.tree import DecisionTreeClassifier
    from sklearn.model_selection import train_test_split
    from sklearn.metrics import accuracy_score, precision_score, recall_score

    # === DATA ===
    X, y = make_classification(n_samples=500, n_features=10, random_state=RANDOM_STATE)
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=TEST_SIZE, random_state=RANDOM_STATE)

    # === MODEL ===
    model = DecisionTreeClassifier(max_depth=MAX_DEPTH, min_samples_split=MIN_SAMPLES_SPLIT, random_state=RANDOM_STATE)
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)

    # === METRICS ===
    metrics = {
        "accuracy": round(accuracy_score(y_test, y_pred), 4),
        "precision": round(precision_score(y_test, y_pred), 4),
        "recall": round(recall_score(y_test, y_pred), 4),
    }

    # === PLOTS ===
    fig, ax = plt.subplots(figsize=(8, 5))
    importances = model.feature_importances_
    ax.barh(range(len(importances)), importances, color='white', edgecolor='gray')
    ax.set_xlabel('Importance')
    ax.set_title('Feature Importances')
    ax.set_facecolor('black')
    fig.patch.set_facecolor('black')
    ax.tick_params(colors='white')
    ax.xaxis.label.set_color('white')
    ax.title.set_color('white')
    buf = BytesIO()
    plt.savefig(buf, format='png', bbox_inches='tight', facecolor='black')
    buf.seek(0)
    plot_b64 = base64.b64encode(buf.read()).decode()
    plt.close()

    result = {
        "metrics": metrics,
        "plots": {"feature_importances": plot_b64},
        "controls": [
            {"id": "max_depth", "type": "slider", "label": "Max Depth", "min": 1, "max": 20, "default": MAX_DEPTH, "step": 1, "targets_var": "MAX_DEPTH"},
            {"id": "min_samples_split", "type": "slider", "label": "Min Samples Split", "min": 2, "max": 20, "default": MIN_SAMPLES_SPLIT, "step": 1, "targets_var": "MIN_SAMPLES_SPLIT"},
            {"id": "test_size", "type": "slider", "label": "Test Size", "min": 0.1, "max": 0.5, "default": TEST_SIZE, "step": 0.05, "targets_var": "TEST_SIZE"},
        ],
        "errors": [],
    }
    print(json.dumps(result))

except Exception as e:
    print(json.dumps({"metrics": {}, "plots": {}, "controls": [], "errors": [str(e)]}))
