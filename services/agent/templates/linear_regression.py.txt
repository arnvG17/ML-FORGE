import json
import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
from io import BytesIO
import base64

try:
    # === TUNABLE PARAMETERS ===
    LEARNING_RATE = 0.01
    N_ITERATIONS = 1000
    RANDOM_STATE = 42
    TEST_SIZE = 0.2

    from sklearn.datasets import make_regression
    from sklearn.linear_model import LinearRegression
    from sklearn.model_selection import train_test_split
    from sklearn.metrics import mean_squared_error, r2_score

    # === DATA ===
    X, y = make_regression(n_samples=200, n_features=1, noise=20, random_state=RANDOM_STATE)
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=TEST_SIZE, random_state=RANDOM_STATE)

    # === MODEL ===
    model = LinearRegression()
    model.fit(X_train, y_train)
    y_pred = model.predict(X_test)

    # === METRICS ===
    metrics = {
        "mse": round(mean_squared_error(y_test, y_pred), 4),
        "r2_score": round(r2_score(y_test, y_pred), 4),
        "coefficient": round(float(model.coef_[0]), 4),
        "intercept": round(float(model.intercept_), 4),
    }

    # === PLOTS ===
    fig, ax = plt.subplots(figsize=(8, 5))
    ax.scatter(X_test, y_test, color='gray', alpha=0.6, label='Actual', edgecolors='white', linewidths=0.5)
    ax.plot(X_test, y_pred, color='white', linewidth=2, label='Predicted')
    ax.set_xlabel('X')
    ax.set_ylabel('y')
    ax.set_title('Linear Regression')
    ax.legend()
    ax.set_facecolor('black')
    fig.patch.set_facecolor('black')
    ax.tick_params(colors='white')
    ax.xaxis.label.set_color('white')
    ax.yaxis.label.set_color('white')
    ax.title.set_color('white')
    ax.legend(facecolor='black', edgecolor='white', labelcolor='white')
    buf = BytesIO()
    plt.savefig(buf, format='png', bbox_inches='tight', facecolor='black')
    buf.seek(0)
    plot_b64 = base64.b64encode(buf.read()).decode()
    plt.close()

    result = {
        "metrics": metrics,
        "plots": {"regression": plot_b64},
        "controls": [
            {"id": "test_size", "type": "slider", "label": "Test Size", "min": 0.1, "max": 0.5, "default": TEST_SIZE, "step": 0.05, "targets_var": "TEST_SIZE"},
        ],
        "errors": [],
    }
    print(json.dumps(result))

except Exception as e:
    print(json.dumps({"metrics": {}, "plots": {}, "controls": [], "errors": [str(e)]}))
