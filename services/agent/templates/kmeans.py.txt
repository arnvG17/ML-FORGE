import json
import numpy as np
import matplotlib
matplotlib.use('Agg')
import matplotlib.pyplot as plt
from io import BytesIO
import base64

try:
    # === TUNABLE PARAMETERS ===
    N_CLUSTERS = 3
    MAX_ITER = 300
    RANDOM_STATE = 42
    N_SAMPLES = 500

    from sklearn.datasets import make_blobs
    from sklearn.cluster import KMeans
    from sklearn.metrics import silhouette_score

    # === DATA ===
    X, y_true = make_blobs(n_samples=N_SAMPLES, centers=N_CLUSTERS, random_state=RANDOM_STATE)

    # === MODEL ===
    model = KMeans(n_clusters=N_CLUSTERS, max_iter=MAX_ITER, random_state=RANDOM_STATE, n_init=10)
    labels = model.fit_predict(X)

    # === METRICS ===
    metrics = {
        "silhouette_score": round(silhouette_score(X, labels), 4),
        "inertia": round(model.inertia_, 2),
        "n_clusters": N_CLUSTERS,
    }

    # === PLOTS ===
    fig, ax = plt.subplots(figsize=(8, 6))
    scatter = ax.scatter(X[:, 0], X[:, 1], c=labels, cmap='gray', alpha=0.6, edgecolors='white', linewidths=0.5)
    centers = model.cluster_centers_
    ax.scatter(centers[:, 0], centers[:, 1], c='white', s=200, marker='x', linewidths=3)
    ax.set_title('K-Means Clusters')
    ax.set_facecolor('black')
    fig.patch.set_facecolor('black')
    ax.tick_params(colors='white')
    ax.title.set_color('white')
    buf = BytesIO()
    plt.savefig(buf, format='png', bbox_inches='tight', facecolor='black')
    buf.seek(0)
    plot_b64 = base64.b64encode(buf.read()).decode()
    plt.close()

    result = {
        "metrics": metrics,
        "plots": {"clusters": plot_b64},
        "controls": [
            {"id": "n_clusters", "type": "slider", "label": "Number of Clusters", "min": 2, "max": 10, "default": N_CLUSTERS, "step": 1, "targets_var": "N_CLUSTERS"},
            {"id": "max_iter", "type": "slider", "label": "Max Iterations", "min": 100, "max": 1000, "default": MAX_ITER, "step": 50, "targets_var": "MAX_ITER"},
        ],
        "errors": [],
    }
    print(json.dumps(result))

except Exception as e:
    print(json.dumps({"metrics": {}, "plots": {}, "controls": [], "errors": [str(e)]}))
